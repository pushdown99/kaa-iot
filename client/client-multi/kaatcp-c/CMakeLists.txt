cmake_minimum_required(VERSION 2.8.8)

project (kaatcp-c)
enable_language(C)

if (CMAKE_COMPILER_IS_GNUCC)
    set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -Os -Wall -std=c99")
endif ()

include_directories (${CMAKE_CURRENT_SOURCE_DIR}/src
)

SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -fPIC")

set (KAA_SOURCE_FILES
                       src/kaatcp_request.c
                       src/kaatcp_parser.c
)

add_library(kaatcp_o OBJECT ${KAA_SOURCE_FILES})

add_library (kaatcp SHARED ./src $<TARGET_OBJECTS:kaatcp_o>)
add_library (kaatcp_s STATIC ./src $<TARGET_OBJECTS:kaatcp_o>)

install (TARGETS kaatcp kaatcp_s
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    RUNTIME DESTINATION lib)

install (DIRECTORY src/ DESTINATION include/kaatcp
    FILES_MATCHING PATTERN *.h)
